async function buscarSpotify() { const artista = document.getElementById('artista').value.trim(); const anyTall = parseInt(document.getElementById('any-tall').value) || 2018; if (!artista) { mostrarStatus('❌ Omple el nom de l\'artista', 'error'); return; } mostrarStatus('⏳ Buscant a Spotify...', 'info'); try { const clientId = localStorage.getItem('spotify_client_id') || ''; const clientSecret = localStorage.getItem('spotify_client_secret') || ''; if (!clientId || !clientSecret) { const id = prompt('Spotify Client ID:'); const secret = prompt('Spotify Client Secret:'); if (id) localStorage.setItem('spotify_client_id', id); if (secret) localStorage.setItem('spotify_client_secret', secret); mostrarStatus('⏳ Token guardat, clica de nou el botó', 'info'); return; } // Token const creds = btoa(clientId + ':' + clientSecret); const tokenResp = await fetch('https://accounts.spotify.com/api/token', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded', 'Authorization': 'Basic ' + creds }, body: 'grant_type=client_credentials' }); const tokenJson = await tokenResp.json(); const token = tokenJson.access_token; if (!token) { mostrarStatus('❌ Error token Spotify — comprova les credencials', 'error'); return; } mostrarStatus('⏳ Buscant artista...', 'info'); // Buscar artista const srResp = await fetch(`https://api.spotify.com/v1/search?q=${encodeURIComponent(artista)}&type=artist&limit=1`, { headers: { 'Authorization': `Bearer ${token}` } }); const srJson = await srResp.json(); const artistItem = srJson.artists?.items[0]; if (!artistItem) { mostrarStatus('❌ Artista no trobat a Spotify', 'error'); return; } const artistId = artistItem.id; mostrarStatus('⏳ Carregant àlbums...', 'info'); // Obtenir àlbums amb any const albumsResp = await fetch( `https://api.spotify.com/v1/artists/${artistId}/albums?include_groups=single,album&market=ES&limit=50`, { headers: { 'Authorization': `Bearer ${token}` } } ); const albumsJson = await albumsResp.json(); const albums = albumsJson.items || []; // Mapa: nom cançó normalitzat -> any const anyPerCanco = {}; const popularitat = {}; // Top tracks per tenir popularitat const topResp = await fetch( `https://api.spotify.com/v1/artists/${artistId}/top-tracks?market=ES`, { headers: { 'Authorization': `Bearer ${token}` } } ); const topJson = await topResp.json(); const topTracks = topJson.tracks || []; const topTracks = topJson.tracks || []; alert('Tracks: ' + topTracks.length + ' | Ex: ' + JSON.stringify(topTracks[0] ? {nom: topTracks[0].name, release: topTracks[0].album?.release_date} : 'buit')); for (const t of topTracks) { const key = t.name.toLowerCase().trim(); const any = parseInt(t.album?.release_date?.split('-')[0]) || 0; anyPerCanco[key] = any; popularitat[key] = t.popularity; } // Completar anys des dels àlbums for (const album of albums.slice(0, 20)) { const any = parseInt(album.release_date?.split('-')[0]) || 0; const atResp = await fetch( `https://api.spotify.com/v1/albums/${album.id}/tracks?limit=50`, { headers: { 'Authorization': `Bearer ${token}` } } ); const atJson = await atResp.json(); for (const t of atJson.items || []) { const key = t.name.toLowerCase().trim(); if (!anyPerCanco[key]) anyPerCanco[key] = any; } } // Construir llista de top tracks amb any correcte const then_list = []; const now_list = []; for (const t of topTracks) { const key = t.name.toLowerCase().trim(); const any = anyPerCanco[key] || parseInt(t.album?.release_date?.split('-')[0]) || 0; if (any === 0) continue; const obj = { nom: t.name, any, era: any < anyTall ? 'THEN' : 'NOW', streams: '', timestamp_manual: null, nom_manual: null }; if (obj.era === 'THEN') then_list.push(obj); else now_list.push(obj); } // Intercalar fins a 10 tracksData = []; let pos = 1; const max = Math.max(then_list.length, now_list.length); for (let i = 0; i < max && tracksData.length < 10; i++) { if (i < then_list.length && tracksData.length < 10) tracksData.push({...then_list[i], pos: pos++}); if (i < now_list.length && tracksData.length < 10) tracksData.push({...now_list[i], pos: pos++}); } if (tracksData.length === 0) { mostrarStatus(`❌ No s'han trobat cançons. Prova un any de tall diferent (actual: ${anyTall})`, 'error'); return; } alert(JSON.stringify(topTracks.slice(0,2).map(t => ({nom: t.name, album: t.album?.name, release: t.album?.release_date})))); renderTracks(); mostrarStatus(`✅ ${tracksData.length} cançons! (${then_list.length} THEN · ${now_list.length} NOW)`, 'ok'); } catch(e) { console.error(e); mostrarStatus('❌ Error: ' + e.message, 'error'); } }
